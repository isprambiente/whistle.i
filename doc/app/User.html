def add_key
  @adding_key = true
  rsa_key = OpenSSL::PKey::RSA.new(2048)
  cipher =  OpenSSL::Cipher::Cipher.new('des3')

    self.priv = rsa_key.to_pem(cipher, password)
    self.pub = rsa_key.public_key.to_pem

  save
end
